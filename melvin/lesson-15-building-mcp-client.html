1. Client Architecture -- Kien truc Client

MCP Client la cau noi giao tiep giua ung dung cua ban va MCP Server. No quan ly message exchange va protocol details de ung dung khong can lam viec do.

Client Architecture
Your App --> MCPClient --> ClientSession --> MCP Server
(custom class)          (MCP SDK)             (external)

Hai thanh phan chinh

1
MCPClient -- Lop custom ma chung ta tao de su dung session de dang hon. Wrapper giup code sach va de dung.
2
ClientSession -- Ket noi thuc te den server, la mot phan cua MCP Python SDK. Xu ly JSON-RPC message exchange.
Nghep cua MCP Client nhu mot "thong dich vien" -- ung dung cua ban noi mot ngon ngu, MCP Server noi ngon ngu khac. Client dich chinh xac cac yeu cau va phan hoi qua giao thuc JSON-RPC.

2. Connection Lifecycle -- Vong doi Ket noi

Client su dung AsyncExitStack de quan ly lifecycle cua nhieu async resources -- dam bao cleanup dung cach khi ket thuc.

Quy trinh ket noi

1
StdioServerParameters -- Dinh nghia cach khoi dong server (command, args, env).
2
stdio_client() -- Tao stdio transport, khoi dong server nhu subprocess.
3
ClientSession(stdio, write) -- Tao session qua read/write streams.
4
session.initialize() -- Gui initialize request, hoan tat handshake voi server.
class MCPClient:
    def __init__(self, command: str, args: list[str], env: Optional[dict] = None):
        self._command = command
        self._args = args
        self._env = env
        self._session: Optional[ClientSession] = None
        self._exit_stack: AsyncExitStack = AsyncExitStack()

    async def connect(self):
        server_params = StdioServerParameters(
            command=self._command,
            args=self._args,
            env=self._env,
        )
        stdio_transport = await self._exit_stack.enter_async_context(
            stdio_client(server_params)
        )
        _stdio, _write = stdio_transport
        self._session = await self._exit_stack.enter_async_context(
            ClientSession(_stdio, _write)
        )
        await self._session.initialize()

Luu y ve read_resource()

Phuong thuc nay kiem tra MIME type: JSON duoc parse thanh Python objects, con lai tra ve raw text. Day la vi du ve logic xu ly ma client can tu viet.

4. Complete Client Code -- Toan bo mcp_client.py

import sys
import asyncio
import json
from typing import Optional, Any
from contextlib import AsyncExitStack
from mcp import ClientSession, StdioServerParameters, types
from mcp.client.stdio import stdio_client
from pydantic import AnyUrl


class MCPClient:
    def __init__(
        self,
        command: str,
        args: list[str],
        env: Optional[dict] = None,
    ):
        self._command = command
        self._args = args
        self._env = env
        self._session: Optional[ClientSession] = None
        self._exit_stack: AsyncExitStack = AsyncExitStack()

    async def connect(self):
        server_params = StdioServerParameters(
            command=self._command,
            args=self._args,
            env=self._env,
        )
        stdio_transport = await self._exit_stack.enter_async_context(
            stdio_client(server_params)
        )
        _stdio, _write = stdio_transport
        self._session = await self._exit_stack.enter_async_context(
            ClientSession(_stdio, _write)
        )
        await self._session.initialize()

    def session(self) -> ClientSession:
        if self._session is None:
            raise ConnectionError(
                "Client session not initialized. Call connect first."
            )
        return self._session

    async def list_tools(self) -> list[types.Tool]:
        result = await self.session().list_tools()
        return result.tools

    async def call_tool(
        self, tool_name: str, tool_input
    ) -> types.CallToolResult | None:
        return await self.session().call_tool(tool_name, tool_input)

    async def list_prompts(self) -> list[types.Prompt]:
        result = await self.session().list_prompts()
        return result.prompts

    async def get_prompt(self, prompt_name, args: dict[str, str]):
        result = await self.session().get_prompt(prompt_name, args)
        return result.messages

    async def read_resource(self, uri: str) -> Any:
        result = await self.session().read_resource(AnyUrl(uri))
        resource = result.contents[0]

        if isinstance(resource, types.TextResourceContents):
            if resource.mimeType == "application/json":
                return json.loads(resource.text)

            return resource.text

    async def cleanup(self):
        await self._exit_stack.aclose()
        self._session = None

    async def __aenter__(self):
        await self.connect()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.cleanup()

5. AsyncExitStack Pattern

AsyncExitStack la mot context manager dac biet cho phep quan ly nhieu async resources cung luc.

Tai sao can AsyncExitStack?

Client can mo nhieu resources theo thu tu: stdio transport, ClientSession. Moi resource can duoc cleanup dung cach khi ket thuc. AsyncExitStack dam bao tat ca duoc dong dung thu tu.

# Trong connect():
self._exit_stack = AsyncExitStack()

# Them tung resource vao stack
stdio_transport = await self._exit_stack.enter_async_context(
    stdio_client(server_params)     # Resource 1: stdio transport
)
self._session = await self._exit_stack.enter_async_context(
    ClientSession(_stdio, _write)   # Resource 2: ClientSession
)

# Cleanup tat ca khi thoat
async def cleanup(self):
    await self._exit_stack.aclose()  # Dong tat ca resources theo thu tu nguoc
    self._session = None

__aenter__ va __aexit__ -- Async Context Manager

Lop MCPClient ho tro async with syntax:

async with MCPClient(command="uv", args=["run", "mcp_server.py"]) as client:
    tools = await client.list_tools()
    # ... su dung client
# Tu dong cleanup khi thoat khoi block

Khi vao block: __aenter__ goi connect(). Khi ra khoi block: __aexit__ goi cleanup(). Khong can nho goi cleanup thu cong.

Client Lifecycle Animation

1

6

Cleanup -- Don dep Resources

Dong tat ca async resources theo thu tu nguoc.

async def cleanup(self):
    await self._exit_stack.aclose()  # Dong ClientSession, roi stdio transport
    self._session = None

# Hoac tu dong khi thoat khoi "async with" block:
# __aexit__ tu dong goi cleanup()
Behind the scenes

AsyncExitStack.aclose() dong cac resources theo thu tu nguoc voi thu tu da dang ky: ClientSession truoc, stdio_transport sau. Server subprocess cung duoc terminated.
